analyzeResume()

function analyzeResume(
  content: string,
  jobKeywords?: string[]
): ResumeAnalysis

Analyzes resume content and extracts structured information including contact details, skills, experience, and generates improvement recommendations.

Parameters

  • content: Raw resume text
  • jobKeywords: Optional job keywords

Returns

  • Complete ResumeAnalysis object
  • Includes strengths, weaknesses, recommendations

scoreResume()

function scoreResume(
  analysis: ResumeAnalysis,
  jobKeywords: string[]
): ScoreResult

Calculates comprehensive match score using weighted algorithm based on skills, experience, completeness, and ATS formatting.

Example Usage

const analysis = analyzeResume(resumeText, jobKeywords);
const score = scoreResume(analysis, jobKeywords);

üîó Integration Examples

Basic Integration

import { analyzeResume, scoreResume, extractKeywords } from './utils/resumeAnalyzer';

// 1. Extract job requirements
const jobDescription = "React developer with 3+ years experience...";
const jobKeywords = extractKeywords(jobDescription);

// 2. Analyze resume
const resumeText = "John Doe, React Developer...";
const analysis = analyzeResume(resumeText, jobKeywords);

// 3. Calculate score
const score = scoreResume(analysis, jobKeywords);

// 4. Make decision
const isQualified = score.percentage >= 60;

Batch Processing

// Process multiple resumes efficiently
const processResumes = async (resumes, jobDesc) => {
  const keywords = extractKeywords(jobDesc);
  const results = [];
  
  for (const resume of resumes) {
    const analysis = analyzeResume(resume.content, keywords);
    const score = scoreResume(analysis, keywords);
    
    results.push({
      ...resume,
      score: score.percentage,
      status: score.percentage >= 60 ? 'qualified' : 'rejected',
      analysis,
      recommendations: analysis.recommendations
    });
  }
  
  return results.sort((a, b) => b.score - a.score);
};

Custom Workflow

// Advanced screening with custom categories
const screenCandidates = (resumes, jobPosting) => {
  const keywords = extractKeywords(jobPosting.description);
  
  return resumes.map(resume => {
    const analysis = analyzeResume(resume.content, keywords);
    const score = scoreResume(analysis, keywords);
    
    // Custom categorization
    let category = 'rejected';
    if (score.percentage >= 80) category = 'priority';
    else if (score.percentage >= 60) category = 'qualified';
    else if (score.percentage >= 40) category = 'potential';
    
    return { ...resume, score, category, analysis };
  });
};
